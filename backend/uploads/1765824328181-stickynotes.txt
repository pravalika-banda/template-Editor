// ======= NOTIFICATIONS PERMISSION =======
async function ensureNotificationPermission() {
  if (!("Notification" in window)) {
    log("Notification API not supported.");
    return false;
  }
  if (Notification.permission === "granted") return true;
  if (Notification.permission === "denied") return false;
  try {
    const perm = await Notification.requestPermission();
    return perm === "granted";
  } catch (e) {
    return false;
  }
}

// ======= RENDER / CREATE NOTE ELEMENT =======
function createNoteElement(note) {
  const div = document.createElement("div");
  div.dataset.id = note.id;
  div.classList.add("note");
if (note.hidden) {
  div.classList.add("hidden");
} else {
  div.classList.add("visible");
}
 div.style.background = note.color || "#fff799";
  div.style.setProperty("--rotate", Math.random());

  const reminderHTML = note.reminder
    ? `<small class="reminder">‚è∞ ${new Date(note.reminder).toLocaleString()}</small>`
    : "";

  div.innerHTML = `
    <div class="note-btns">
      <button onclick="editNote(${note.id})">‚úèÔ∏è</button>
      <button onclick="toggleHide(${note.id})">üëÅÔ∏è</button>
      <button onclick="deleteNote(${note.id})">üóëÔ∏è</button>
    </div>
    <p>${escapeHtml(note.text)}</p>
    ${reminderHTML}
    <small class="created">üïí ${note.time || ""}</small>
  `;

  if (note.reminder && !note.reminded) {
    const now = new Date();
    const r = new Date(note.reminder);
    if (r <= now) div.classList.add("reminder-active");
  }

  return div;
}

function renderAllNotes() {
  board.innerHTML = "";
  notes.forEach(n => board.appendChild(createNoteElement(n)));
}

// escape html to avoid injection when rendering user text
function escapeHtml(s) {
  return s.replace(/[&<>"']/g, c => ({'&':'&amp;','<':'&lt;','>':'&gt;','"':'&quot;',"'":'&#39;'}[c]));
}

// ======= SCHEDULE / TRIGGER REMINDERS =======
function scheduleReminder(note) {
  // clear previous timer if exists
  if (scheduledTimers.has(note.id)) {
    clearTimeout(scheduledTimers.get(note.id));
    scheduledTimers.delete(note.id);
  }
  if (!note.reminder || note.reminded) return;

  const remindAt = new Date(note.reminder).getTime();
  if (isNaN(remindAt)) {
    log("Invalid reminder for note", note.id, note.reminder);
    return;
  }

  const now = Date.now();
  const delay = remindAt - now;

  if (delay <= 0) {
    // already due ‚Äî trigger immediately (but don't block UI)
    triggerReminder(note);
    return;
  }

  const t = setTimeout(() => {
    triggerReminder(note);
    scheduledTimers.delete(note.id);
  }, delay);

  scheduledTimers.set(note.id, t);
  log(`Scheduled reminder for note ${note.id} in ${Math.round(delay/1000)}s`);
}

async function triggerReminder(note) {
  // mark visually first
  renderAllNotes();

  const hasPerm = await ensureNotificationPermission();

  if (hasPerm) {
    try {
      new Notification("Sticky Notes Reminder", {
        body: note.text,
        icon: "https://cdn-icons-png.flaticon.com/512/1827/1827349.png",
      });
    } catch (e) {
      log("Notification error:", e);
    }
  } else {
    // fallback
    alert(`‚è∞ Reminder: ${note.text}`);
  }

  // visual pulse on the specific note element
  const nodes = Array.from(document.querySelectorAll(".note"));
  const idx = notes.findIndex(n => n.id === note.id);
  if (nodes[idx]) {
    nodes[idx].classList.add("reminder-active");
    // remove pulse after a while (optional)
    setTimeout(() => nodes[idx] && nodes[idx].classList.remove("reminder-active"), 15000);
  }

  // mark as reminded and save
  note.reminded = true;
  saveNotes();
  renderAllNotes();
}

// periodic fallback check (in case setTimeouts were lost or missed)
setInterval(() => {
  const now = Date.now();
  notes.forEach(n => {
    if (n.reminder && !n.reminded) {
      const remindAt = new Date(n.reminder).getTime();
      if (!isNaN(remindAt) && remindAt <= now) {
        log("Fallback trigger for note", n.id);
        triggerReminder(n);
      }
    }
  });
}, 10000); // check every 10s

// ======= CRUD: add / edit / delete / hide =======
addBtn.onclick = async () => {
  const text = noteText.value.trim();
  if (!text) return alert("Please write a note first!");

  const currentTime = new Date().toLocaleString();
  const note = {
    id: Date.now(),
    text,
    color: noteColor.value || "#fff799",
    reminder: reminderInput.value || null,
    reminded: false,
    hidden: false,
    time: currentTime
  };

  notes.push(note);
  saveNotes();
  renderAllNotes();

  // schedule immediately
  scheduleReminder(note);

  // clear inputs
  noteText.value = "";
  reminderInput.value = "";
};

function editNote(id) {
  const note = findNoteById(id);
  if (!note) return;
  const newText = prompt("Edit your note:", note.text);
  if (newText === null) return;
  if (newText.trim() !== note.text) {
    if (confirm("Save changes?")) {
      note.text = newText.trim();
      saveNotes();
      renderAllNotes();
    }
  }
}

function toggleHide(id) {
  const note = findNoteById(id);
  if (!note) return;

  // Set the current note as target
  targetNote = note;

  // Show modal with the correct message
  confirmMsg.textContent = note.hidden
    ? "Do you want to unhide this note?"
    : "Are you sure you want to hide this note?";

  modal.style.display = "flex";
}
// ======= Modal Buttons Logic =======
yesBtn.addEventListener("click", () => {
  if (targetNote) {
    // Toggle hide/unhide
    targetNote.hidden = !targetNote.hidden;
    saveNotes();

    const noteDiv = document.querySelector(`[data-id="${targetNote.id}"]`);
    if (noteDiv) {
      noteDiv.classList.toggle("hidden", targetNote.hidden);
      noteDiv.classList.toggle("visible", !targetNote.hidden);
    } else {
      renderAllNotes(); // fallback
    }

    // Optional visual confirmation
    alert(targetNote.hidden ? "‚úÖ Note hidden successfully!" : "üëÅÔ∏è Note is now visible again!");
  }

  modal.style.display = "none";
  targetNote = null;
});

cancelBtn.addEventListener("click", () => {
  modal.style.display = "none";
  targetNote = null;
});

// Hide modal if background is clicked
window.addEventListener("click", (e) => {
  if (e.target === modal) {
    modal.style.display = "none";
    targetNote = null;
  }
});

function deleteNote(id) {
  if (!confirm("Delete this note?")) return;
  notes = notes.filter(n => n.id !== id);
  saveNotes();
  renderAllNotes();
}

// ======= ON LOAD: render and schedule reminders for saved notes =======
(async function init() {
  renderAllNotes();
  // ask permission early (optional) ‚Äî will show browser prompt
  ensureNotificationPermission().then(p => log("Notification permission:", p));
  // schedule reminders for saved notes
  notes.forEach(n => {
    if (n.reminder && !n.reminded) scheduleReminder(n);
  });
})();

// ======= helper: saveNotes =======
function saveNotes() {
  localStorage.setItem("notes", JSON.stringify(notes));
}
